var EDITOR = {
    /**
     * 에디터 생성
     *
     * @method createEditor
     * @param {object}   container container
     * @param {object}   [options]              에디터 생성 옵션
     * @param {string}   [options.width="100%"] 가로크기
     * @param {string}   [options.height="100%"] 세로크기
     * @param {boolean}  [options.editable=true] 편집모드 여부
     * @return {object} Promise
     * @async
     * @example
     *     initEditor: function() {
     *         var me = this;
     *         var options = {
     *             editable: me.get("editable") === true || me.get("editable") === "true"
     *         };
     *         return EDITOR.createEditor(me.$.editorPanel, options);
     *     }
     */
    createEditor: function(container, options) {
        return new Promise(function(resolve, reject) {
            options = options || {};

            // DEXT5.config.InitXml = "dext_editor.xml";
            // DEXT5.config.HandlerUrl =
            // "http://www.dext5.com/handler/upload_handler.ashx";
            // DEXT5.config.SkinName = "blue";
            // DEXT5.config.ToSavePathURL = "dext5data";
            var lang = "ko-kr";
            DEXT5.config.Lang = lang; // ko-kr, en-us, ja-jp, zh-cn, zh-tw
            // DEXT5.config.focusInitObjId = "title"; // 에디터가 로드된 후 설정한 아이디를 가진 객체에
            // 포커스를 줌.
            // DEXT5.config.NextTabElementId = "nextInput";
            // DEXT5.config.IgnoreSameEditorName = "1";
            var editorId = container.contentId;
            DEXT5.config.Mode = (UT.isBoolean(options.editable) && options.editable === false) ? "view" : "edit";
            DEXT5.config.Width = options.width || "100%";
            DEXT5.config.Height = options.height || "100%";
            DEXT5.config.EditorHolder = editorId;
            // 에디터의 팝업창과 우클릭 메뉴를 부모창에 띄우기 위한 설정.
            // 에디터를 iframe 형태로 띄우는 경우 사이즈가 작아서 팝업창이나 우클릭 메뉴가 가려질 때 설정하면 유용합니다.
            DEXT5.config.DialogWindow = parent.window;
            DEXT5.config.LoadedEvent = function() { // 에디터가 로드 완료(작업준비)
                EDITOR.getBodyValue(editorId, "");
                resolve({resolveName: "editor-created", editorId: editorId});
            };
            new Dext5editor(editorId);
        });
    },

    /**
     * Html 값 지정 - &lt;html> 태그 내부의 소스

     * @param {boolean} editable editable
     * @example
