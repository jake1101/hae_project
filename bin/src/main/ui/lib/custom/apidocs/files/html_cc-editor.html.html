/**
 * 에디터 커스텀 태그
 *
 *     <cc-editor id="editor" class="h-400" value="{{value}}" value-type="body" editable="{{editable}}"></cc-editor>
 *
 * @element cc-editor
 *
 *
 * @attribute value 에디터에 삽입될 내용 binding value <br>
 * [string] [optional] 
 *
 *
 * @attribute value-type value 유형 <br>
 * [string] [optional] default="body" <br>
 *     body    : <body> 태그 내부의 소스
 *     body-ex : <body> 부터 </body> 까지의 소스 
 *     html    : <html> 태그 내부의 소스 
 *     html-ex : <html> 부터 </html>까지의 소스
 *
 *
 * @attribute editable 편집모드 설정 <br>
 * [boolean] [optional] default=false
 *
 *
 * @example
 *     // methods
 *     * syncValue() - 에디터 내용과 binding value의 값을 동기화한다. 
 *         someFunction: function() {
 *             var me = this;
 *             me.$.editor.syncValue();
 *             ... 
 *         } 
 */

<dom-module id="cc-editor">
    <style>
    </style>
    
    <template>
        <sc-container id="editorPanel" class$="[[class]]"  flexible="[[flexible]]"></sc-container>
    </template>
    
    <script>
        Polymer({
            is: "cc-editor",
            properties: {
                editorId: {
                    type: String,
                    value: "",
                    notify: true
                },
                editable: {
                    type: Object,
                    value: function() {
                        return false;
                    },
                    observer: "changedEditable"
                },
                valueType: {
                    type: String,
                    value: "body" // body, body-ex, html, html-ex 
                },
                value: {
                    type: String,
                    value: "",
                    notify: true,
                    observer: "changedValue"
                }
            },
            
            // 화면 생성 완료
            attached: function() {
                var me = this;
            },

            // editor service
            service: function() {
                var me = this;
                var promise = [
                    me.initEditor()
                ];
                Promise.all(promise).then(function(res) {
                    me.set("editorId", res[0].editorId);
                    me.fire("result", res[0].resolveName);
                });
            },
            
            // 에디터 생성
            initEditor: function() {
                var me = this;
                var options = {
                    editable: me.get("editable") === true || me.get("editable") === "true"
                };
                return EDITOR.createEditor(me.$.editorPanel, options);
            },
            
            // 편집모드 설정
            changedEditable: function(editable) {
                var me = this;
                if (me.editorId) {
                    EDITOR.setEditable(me.editorId, editable === true || editable === "true");
                }
            },
            
            // value 값 설정
            changedValue: function(value) {
                var me = this;
                if (me.editorId && typeof value === "string") {
                    switch(me.get("valueType")) {
                    case "html-ex": // <html> 부터 </html>까지의 소스
                        EDITOR.setHtmlValueEx(me.editorId, value);
                        break;
                    case "body-ex": // <body> 부터 </body> 까지의 소스
                        EDITOR.setBodyValueEx(me.editorId, value);
                        break;
                    case "html": // <html> 태그 내부의 소스
                        EDITOR.setHtmlValue(me.editorId, value);
                        break;
                    default:     // <body> 태그 내부의 소스
                        EDITOR.setBodyValue(me.editorId, value);
                        break;
                    }
                }
            },
            
            // 에디터 값과 bind 변수값을 동기화한다. 
            syncValue: function() {
                var me = this;
                if (me.editorId) {
                    switch(me.get("valueType")) {
                    case "html-ex":
                        me.set("value", EDITOR.getHtmlValueEx(me.editorId));
                        break;
                    case "body-ex":
                        me.set("value", EDITOR.getBodyValueEx(me.editorId));
                        break;
                    case "html":
                        me.set("value", EDITOR.getHtmlValue(me.editorId));
                        break;
                    default:
                        me.set("value", EDITOR.getBodyValue(me.editorId));
                        break;
                    }
                }
            },
            
            // 현재 커서 위치에 html 삽입
            setInsertHtml: function(html) {
                var me = this;
                if (me.editorId) {
                    EDITOR.setInsertHtml(me.editorId, html);
                }
            }
            
        });
    </script>

</dom-module>
