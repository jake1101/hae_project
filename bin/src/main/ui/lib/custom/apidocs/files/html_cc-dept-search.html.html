<link rel="import" href="../../../bp/shared/ep-dept-list.html">

<dom-module id="cc-dept-search">
    <style>
        :host {
            display : table;
        }
        :host > * {
            vertical-align : middle;        
            display: table-cell;
        }    
    </style>
    
    <template>
       <sc-text-field class$="[[class]]" value="{{value}}" placeholder$="[[placeholder]]" on-enter="onFindDept" trigger-cls="search" on-trigger-click="onFindDept" maxLength="200"></sc-text-field>
    </template>
    
    <script>
        Polymer({
            is : 'cc-dept-search',
            properties : {
                value : {
                    type : Object,
                    notify: true,
                    observer : 'valueChanged'
                },
                autoComFlage : {
                    type : Boolean,
                    value : false
                },
                searchResult : {
                    type : Object,
                    value : function(){
                        return {
                            dept_cd : ''
                        };
                    },
                    observer : 'searchResultChanged',
                    notify: true
                },
                resultField : {
                    type : String,
                    value : 'value'
                },
                resultValue : {
                    type : Object,
                    notify: true
                },
                labelDisabled : {
                    type : Object,
                    value : function(){
                        return false;    
                    },
                    observer : 'labelDisabledChanged'
                }
            },
            //최초 호출
            attached : function(){
            },
    
            //label 표시 처리
            labelDisabledChanged : function(newValue, oldValue){
                var me = this;
                if(!newValue){
                    var module = document.createElement('sc-label');
                    Polymer.dom(this.root).appendChild(module);
                }
            },
            
            //결과  변경시 label표시 처리
            searchResultChanged: function(newValue, oldValue) {
                if(this.children[1]){
                    this.children[1].text=newValue.dept_cd;
                }
            },
            
            //값 변경시 result값 clear처리
            valueChanged : function(newValue, oldValue) {
                if(!this.autoComFlage){
                    this.set("searchResult",{});
                    this.set("resultValue",'');
                }
                this.set("autoComFlage",false);
            },
            
            onFindDept : function(){
                var me = this;
                //자동완성 함수 시작
                UT.autoCompleteObject('ep-dept-list', me, {'org_cd' : SCSessionManager.currentUser.comp_cd, 'dept_nm' : me.value}, function(result) {
                    //결과값 처리하는 callback
                    me.set("autoComFlage",true);
                    if(me.resultField){
                        me.set("resultValue",result[me.resultField]);
                    }
                    me.set("searchResult",result);
                    me.set("value",result.dept_nm);
                    me.fire("result",result);
                    me.set("autoComFlage",false);
                },{title : me.translate("부서 검색")});//:~자동완성 함수 끝
            }
        });
    </script>
</dom-module>
