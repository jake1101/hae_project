<!-- 
/**
 * 복수 검색을 위한 input field 커스텀 태그
 *
 *     <cc-multi-text-field value="{{value}}" multi-value="{{여러개받을value}}" class="w-200" title="결재번호"></cc-multi-text-field>
 *
 * @element cc-multi-text-field
 *
 *
 * @attribute value 해당 텍스트 필드 input value binding <br>
 * [string] [optional] 
 *
 *
 * @attribute placeholder placeholder <br>
 * [string] [optional] 
 *
 *
 * @attribute title 다이얼로그 팝업 제목 <br>
 * [string] [optional]
 *
 *  @attribute multi-value 멀티 value<br>
 * [string] [optional]
 *
  *  @attribute multi-check 멀티 체크 초기값 Y or N ,default N<br>
 * [string] [optional]
 *
 */
-->
<dom-module id="cc-multi-text-field">
    <style>
        :host {
            display : table;
        }
        :host > * {
            display: table-cell;
            vertical-align : middle;
        }      
    </style>
    
    <template>

    	<sc-text-field 
            class$="[[class]]"
            value="{{value}}" placeholder$="[[placeholder]]" hidden="{{uiBind.multiable}}" maxLength="200">
        </sc-text-field>
        <sc-text-field 
            class$="[[class]]" trigger-cls="multi-search" readonly
            value="{{displayValue}}" placeholder$="[[placeholder]]" 
            on-trigger-click="showDialog" hidden="{{!uiBind.multiable}}">
        </sc-text-field>
        &nbsp;
		<sc-checkbox-field  
        	on-click="onChangeMultiSelect" 
        	input-value="{{multiCheck}}"
        	checked-value="Y" un-checked-value="N"></sc-checkbox-field>

        <sc-dialog id="dialog" title="{{title}}" title-align="left" style="width:300px;height:400px" modal="true">
            <div style="display:flex;flex-direction:column;height:100%;width:100%;">
                <sc-toolbar>
                    <sc-text-field class="w-50" value="{{rowCount.add}}"></sc-text-field>
                    <sc-button text="#{행추가}" on-click="onAddRows" disabled$="{{!uiBind.addable}}"></sc-button>
                    <sc-spacer></sc-spacer>
                    <sc-button text="#{확인}" on-click="onOk"></sc-button>
                    <sc-button text="#{지우기}" on-click="onClear"></sc-button>
                </sc-toolbar>
                <sc-container id="gridPanel" flexible="true"></sc-container>
            </div>
        </sc-dialog>
    </template>
    
    <script>
        Polymer({
            is: 'cc-multi-text-field',
            properties: {
                value: {
                    type: Object,
                    value : function(){
                    	return {};
                    },
                    notify: true
                },
                multiValue: {
                	type: Array,
                    value: function() {
                        return [];
                    },
                    notify: true,
                    observer: "changedValue"
                },
                displayValue: {
                    type: String,
                    value: ""
                },
                multiCheck : {
					type : Object,
					value : function(){
						return "N";
					},
					notify:true
				},
                userEdit: { // 직접입력 여부
                    type: Boolean,
                    value: false
                },
                initValue: {
                    type: Array,
                    value: function() {
                        return [];
                    }
                },
                rowCount: {
                    type: Object,
                    value: function() {
                        return {
                            max: 200,
                            init: 100,
                            add: 10
                        }
                    }
                },
                uiBind: {
                    type: Object,
                    value: function() {
                        return {
                            addable: true,
                            multiable : false
                        }
                    }
                },
                gridView: Object
            },

            // 화면 생성 완료
            attached: function() {
                var me = this;
                me.initGrid();
                me.setupGrid();
                me.clearGridRows();
                me.changedValue(me.get("initValue"));
            },
            
            // 그리드 생성
            initGrid: function() {
                var me = this;
                var fields = [
                    "item"
                ];
                var columns = [
                    "item | 항목 | 200"
                ];
                me.gridView = GRID.createGridView(me.$.gridPanel, fields, columns, {
                    editableFields: ["item"]
                });
            },

            // 그리드 설정
            setupGrid: function() {
                var me = this;
                GRID.sortable(me.gridView, false); // 소팅기능 없음
                GRID.setFlexibleColumn(me.gridView, "item");
                GRID.setOptions(me.gridView, {
                    stateBar: {
                        visible: false
                    },
                    checkBar: {
                        visible: false
                    }
                });
            },
            
            // 그리드 데이터 초기화
            clearGridRows: function() {
                var me = this;
                var rowCount = Math.max(me.get("rowCount.init"), me.gridView.getDataProvider().getRowCount());
                GRID.clearRows(me.gridView);
                for (var i = 0; i < rowCount; i++) {
                    GRID.addRow(me.gridView, {item: ""});
                }
            },
            
            // 다이얼로그 보이기
            showDialog: function() {
                var me = this;
                me.$.dialog.show();
            },
            
            // 값 변경
            changedValue: function(value) {
                var me = this;
                if (!me.gridView) {
                    me.set("initValue", value);
                    return;
                }
                if (UT.isArray(value)) {
                    for (var i = 0, len = value.length; i < len; i++) {
                        if (value[i]) {
                            me.gridView.setValues(i, {item: value[i]});
                        }
                    }
                    var displayValue = "";
                    if (value[0]) {
                        displayValue = value[0];
                        if (value.length > 1) {
                            displayValue = this.translate("{0}외 {1}건",null,value[0], value.length - 1);
                        }
                    }
                    me.set("displayValue", displayValue);
                }
            },

            // 행추가 버튼
            onAddRows: function() {
                var me = this;
                var maxRowCount = me.get("rowCount.max");
                var addRowCount = me.get("rowCount.add");
                if (isNaN(addRowCount)) {
                    return;
                }
                var addableCount = maxRowCount - me.gridView.getDataProvider().getRowCount();
                
                addRowCount = Math.min(parseInt(addRowCount, 10), addableCount);
                for (var i = 0; i < addRowCount; i++) {
                    GRID.addRow(me.gridView, {item: ""}, "bottom");
                }
                me.set("uiBind.addable", maxRowCount - me.gridView.getDataProvider().getRowCount() > 0);
            },
            
            // 확인 버튼
            onOk: function() {
                var me = this;
                var all = GRID.allRows(me.gridView);
                var items = [];
                for (var i = 0, len = all.length; i < len; i++) {
                    if (!UT.isEmpty(all[i].item)) {
                        items.push(all[i].item);
                    }
                }
                me.set("multiValue", items);
                me.$.dialog.close();
            },
            
            // 지우기 버튼
            onClear: function() {
                var me = this;
                me.clearGridRows();
                me.set("displayValue", "");
                me.set("multiValue", []);
            },
            //********************************************************//
          //다중선택 체크 변경
			onChangeMultiSelect : function(e){
				var me = this;
				me.set("uiBind.multiable",me.multiCheck === "Y");
				me.set("value","");
				me.onClear();
			}
        });
    </script>
</dom-module>
