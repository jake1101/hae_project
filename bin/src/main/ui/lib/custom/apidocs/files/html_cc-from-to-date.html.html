/**
 * from date ~ to date 를 위한 text input 커스텀 태그
 *
 *     <cc-from-to-date from-date="{{from_date}}" to-date={{to_date}}></cc-from-to-date>
 *
 * @element cc-from-to-date
 *
 *
 * @attribute from-date 시작일자 value binding <br>
 * [string] [optional] 
 *
 *
 * @attribute to-date 종료일자 value binding <br>
 * [string] [optional] 
 *
 *
 * @attribute init-from-date fromDate 기본값 설정 <br>
 * (ex) init-from-value="-30" (fromDate로 부터 30일 전으로 fromDate값 세팅)<br>
 * [number] [optional] 
 *
 *
 * @attribute init-to-date toDate 기본값 설정 <br>
 * [number] [optional] <br> 
 * (ex) init-to-value="30" (toDate로 부터 30일 후로 toDate값 세팅)
 *
 *
 * @attribute init-date fromDate, toDate 기본값 세팅 여부 <br>
 * [boolean] [optional] default=true <br>
 * - "true" 인 경우, fromDate = "현재날짜 -7", toDate="현재날짜"로 세팅 됨 <br>
 * - "false"인 경우, fromDate, toDate 값 세팅 안함.
 *
 *
 * @attribute time-zone  sc-date-field의 value값 사용을 지정한다 <br>
 * [boolean] [optional] default=false<br> 
 * - "false"인 경우, sc-date-field의 value를 String 값으로 사용 합니다.
 *
 *
 */
-->

    <template>
            <sc-date-field value="{{fromDate}}" max-date="{{toDate}}" string-date="{{!timeZone}}" required$="[[required]]" disabled$="[[disabled]]" readonly$="[[readonly]]"></sc-date-field>
                &nbsp;~&nbsp;
            <sc-date-field value="{{toDate}}" min-date="{{fromDate}}" string-date="{{!timeZone}}" required$="[[required]]" disabled$="[[disabled]]" readonly$="[[readonly]]"></sc-date-field>
    </template>

<!--  
    ************************************************************************************************************
    * Script Area
    ************************************************************************************************************
-->
    <script>
        Polymer({
            is: "cc-from-to-date",
            properties : {
                 // fromDate 값
                fromDate : {
                    type : String,
                    notify: true
                },
                
                // toDate 값
                toDate : {
                    type : String,
                    notify: true
                },
                
                // fromDate 기본값 설정
                // ex) init-from-value="-30" (fromDate로 부터 30일 전으로 fromDate값 세팅)
                initFromValue : {
                    type : Number
                },
                
                // toDate 기본값 설정
                // ex) init-to-value="30" (toDate로 부터 30일 후로 toDate값 세팅)
                initToValue : {
                    type : Number
                },
                
                // fromDate, toDate 기본값 세팅 여부 ( default = "true" )
                // - "true" 인 경우, fromDate = "현재날짜 -7", toDate="현재날짜"로 세팅 됨
                // - "false"인 경우, fromDate, toDate 값 세팅 안함.
                initDate: {
                    type: Object,
                    value: function(){
                        return true;
                    }
                },
                
                // sc-date-field의 value값 사용을 지정한다. ( default = "false" )
                // - "false"인 경우, sc-date-field의 value를 String 값으로 사용 합니다.
                timeZone: {
                    type: Object,
                    value: function(){
                        return false;
                    }
                }
            },
            
            observers : ["_dateValidCheck(fromDate, toDate)",
                         "changeInitDate(initDate)"],
            
            // 시작일자 또는 종료일자 값 변경시 체크
            // - 시작일자와 종료일자를 비교하여 유효한 기간인지 체크한다.
            _dateValidCheck: function(fromValue, toValue){
                var me = this;
                
                if(fromValue && toValue){
                    // 시작날짜 & 종료날짜 비교 체크
                    var fromDt;
                    var toDt;
                    
                    if(fromValue instanceof Date){
                        var fromDt = fromValue;
                    }else{
                        var fromDt = new Date(fromValue.substr(0,4), fromValue.substr(4,2), fromValue.substr(6,2));
                    }
                    
                    if(toValue instanceof Date){
                        var toDt = toValue;
                    }else{
                        var toDt = new Date(toValue.substr(0,4), toValue.substr(4,2), toValue.substr(6,2));
                    }
                    
                    // fromDt와 toDt의 경과일수를 계산한다.
                    var btMs = fromDt.getTime() - toDt.getTime() ;
                    var btDay = btMs / (1000*60*60*24) ;
                    
                    // 경과일수가 0보다 클 경우
                    if(btDay > 0){
                        UT.alert(this.translate("시작일자가 더 큽니다."));
                    }
                    
                }
            },
            changeInitDate : function(){
            	var me = this;
            	
            	if(me.get("timeZone") === false && me.get("initDate") === true && !me.fromDate && !me.toDate){
                    
                    if(me.initFromValue == null || me.initFromValue =='undefined') {
                        // 기본값은 미 지정 시 일주일 전으로 초기화
                        me.initFromValue = -7;
                    }
                      
                    if(me.initToValue == null || me.initToValue =='undefined') {
                        me.initToValue = 0;
                    }
                      
                    // 시작일자, 종료일자 defaultDate 설정
                    var fromdate1 = new Date();
                    fromdate1 = new Date(Date.parse(fromdate1) + (me.initFromValue) * 1000 * 60 * 60 * 24);
                 
                    var todate1 = new Date();
                    todate1 = new Date(Date.parse(todate1) + (me.initToValue) * 1000 * 60 * 60 * 24);
                    
                    me.fromDate = fromdate1.toISOString().slice(0,10).replace(/-/g,"");
                    me.toDate = todate1.toISOString().slice(0,10).replace(/-/g,"");
                }
            },
            attached: function() {
                var me = this;
                
                //바인딩처리 수정해야함.
                // 1. iniDate값이 true 이고
                // 2. fromDate & toDate 값이 없는 경우, 기본값으로 현재일자 세팅
                if(me.get("initDate") === true && !me.fromDate && !me.toDate){
                    
                    if(me.initFromValue == null || me.initFromValue =='undefined') {
                        // 기본값은 미 지정 시 일주일 전으로 초기화
                        me.initFromValue = -7;
                    }
                      
                    if(me.initToValue == null || me.initToValue =='undefined') {
                        me.initToValue = 0;
                    }
                      
                    // 시작일자, 종료일자 defaultDate 설정
                    var fromdate1 = new Date();
                    fromdate1 = new Date(Date.parse(fromdate1) + (me.initFromValue) * 1000 * 60 * 60 * 24);
                 
                    var todate1 = new Date();
                    todate1 = new Date(Date.parse(todate1) + (me.initToValue) * 1000 * 60 * 60 * 24);
                    
                    me.fromDate = fromdate1.toISOString().slice(0,10).replace(/-/g,"");
                    me.toDate = todate1.toISOString().slice(0,10).replace(/-/g,"");
                }
                
            }
        });
    </script>

</dom-module>
