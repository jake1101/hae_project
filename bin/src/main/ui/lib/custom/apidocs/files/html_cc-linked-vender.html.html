<link rel="import" href="../../../bp/shared/ep-linked-vendor-list.html">

<dom-module id="cc-linked-vender">
    <style>
    </style>
    
    <template>
        <div class="field-box">
            <sc-text-field value="{{vdNm}}" 
                trigger-cls="search" on-trigger-click="onFind" on-enter="onFind" on-keydown="onKeyDown"
                readonly$="[[readonly]]" required$="[[required]]" class$="[[class]]" placeholder="#{협력사명}">
            </sc-text-field>
            <sc-text-field value="{{erpVdCd}}" 
                readonly="true" required$="[[required]]" placeholder="#{코드}" style="margin-left:5px; width:70px !important;">
            </sc-text-field>
        </div>
    </template>
    
    <script>
        Polymer({
            is: "cc-linked-vender",
            properties: {
                vdCd: {
                    type: String,
                    value: "",
                    notify: true
                },
                vdNm: {
                    type: String,
                    value: "",
                    notify: true
                },
                erpVdCd: {
                    type: String,
                    value: "",
                    notify: true
                },
                linkTyp: {
                    type: String,
                    value: ""
                },
                operOrgCd: {
                    type: String,
                    value: ""
                }
            },

            attached: function(){

            },
    
            // 사용자의 keydown
            onKeyDown: function() {
                var me = this;
                console.log(me.readonly)
                if (me.readonly === true) {
                    return;
                }
                me.set("vdCd", "");
                me.set("erpVdCd", "");
            },
            
            // 협력사 찾기
            onFind: function(){
                var me = this;
                if (me.readonly === true) {
                    return;
                }
                if(!me.linkTyp){
                    UT.alert(this.translate("연결유형을 선택하십시오"));
                    return;
                }
                if(!me.operOrgCd){
                    UT.alert(this.translate("운영조직을 선택하십시오"));
                    return;
                }
                var param = {
                    link_typ: me.get("linkTyp"), // 연결유형
                    src_oper_org_cd: me.get("operOrgCd"), // 운영조직
                    erp_vd_cd: me.get("erpVdCd"), // 협력사코드
                    vd_nm: me.get("vdNm") // 협력사명
                };
                UT.autoCompleteObject("ep-linked-vendor-list", me, param, function(result) {
                    me.set("erpVdCd", result.erp_vd_cd);
                    me.set("vdCd", result.vd_cd);
                    me.set("vdNm", result.vd_nm);
                });
            }
        });
    </script>
</dom-module>
