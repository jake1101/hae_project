<link rel="import" href="../../../bp/shared/ep-job-list.html">

<dom-module id="cc-job-search">
    <style>
    </style>
    
    <template>
        <div class="field-box">
            <sc-text-field value="{{display}}" 
                trigger-cls="search" on-trigger-click="onFind" on-enter="onFind" on-keydown="onKeyDown" maxLength="200"
                readonly$="[[readonly]]" class$="[[class]]" placeholder$="[[placeholder]]" required$="[[required]]">
            </sc-text-field>
            <span style="margin:4px 0 0 5px;" hidden$="[[labelDisabled]]">{{value}}</span>
        </div>
    </template>
    
    <script>
        Polymer({
            is: "cc-job-search",
            properties: {
                value: {
                    type: String,
                    value: "",
                    notify: true
                },
                display: {
                    type: String,
                    value: "",
                    notify: true
                },
                valueField: {
                    type: String,
                    value: "purc_grp_cd"
                },
                displayField: {
                    type: String,
                    value: "purc_grp_nm"
                },
                purcGrpTyp: {
                    type: String,
                    value: ""
                },
                labelDisabled : {
                    type : Object,
                    value : function(){
                        return false;    
                    }
                }
            },

            attached: function(){

            },
    
            // 사용자의 keydown
            onKeyDown: function() {
                var me = this;
                if (me.readonly === true) {
                    return;
                }
                me.set("value", "");
            },
            
            // 구매그룹 찾기
            onFind: function(){
                var me = this;
                if (me.readonly === true) {
                    return;
                }
                if(!me.purcGrpTyp){
                    UT.alert(this.translate("직무유형을 선택하십시오"));
                    return;
                }
                var param = {
                    comp_cd: SCSessionManager.currentUser.comp_cd,
                    purc_grp_typ: me.get("purcGrpTyp"), //직무유형
                };
                param[me.displayField] = me.get("display");
                
                UT.autoCompleteObject("ep-job-list", me, param, function(result) {
                    me.set("display", result[me.displayField]);
                    me.set("value", result[me.valueField]);
                },{title : me.translate("구매그룹검색")});
            }
        });
    </script>
</dom-module>
