/**
 * 운영단위 멀티 콤보박스 필드 커스텀 태그
 * ajax통신을 사용하는 custom dom으로 promise로 묶어야함으로 id를 부여 하여 구현한다.
 * service함수를 구현 한다.
 *
 *     <cc-operorg-multi-combobox-field id="operorgcombobox" class="w-150" selected-index="0" value="{{searchParam.oper_org_cd}}" placeholder="전체" oper-unit-cd="PO"></cc-operorg-multi-combobox-field>
 *
 * @element cc-operorg-multi-combobox-field
 *
 *
 * @attribute value 해당 텍스트 필드 input value binding <br>
 * [string] [optional] 
 *
 *
 * @attribute placeholder placeholder <br>
 * [string] [optional] 
 *
 *
 * @attribute enable-check-all 전체 선택 체크박스표시 여부 default값 true<br>
 * [string] [optional]
 *
 *	@attribute disabled disabled <br>
 
  * @attribute oper-unit-cd 운영단위코드
 * [string] [optional]
 *
 *	@attribute disabled disabled <br>
 *
 */
-->

<dom-module id="cc-operorg-multi-combobox-field">
    <style>
        :host {
            display : table;
        }    
        
        :host > * {
            vertical-align : middle;
            display: table-cell;
        }
    </style>
    
    <template>
        <!-- ajax 선언 부 -->
        <!-- 사용자 구매 운영조직 조회 -->
        <sc-service-group id="codes">
            <sc-ajax id="findListOperOrgByUser"
                url="findListOperOrgByUser.do"
                body$="[[operUnitCd]]"
                content-type="text/plain"
                last-response="{{codes.operOrgCd}}">
            </sc-ajax>
          </sc-service-group>
          
       <sc-multi-combobox-field 
                   class$="[[class]]" 
                   selected-all="{{selectedAll}}"
                   enable-check-all="{{enableCheckAll}}"
                   display-field="org_nm" 
                   value-field="oper_org_cd"  
                   items="[[codes.operOrgCd]]" 
                   value = "{{value}}"
                   disabled ="{{disabled}}"
                   required$="[[required]]"
                   placeholder$="[[placeholder]]">
       </sc-multi-combobox-field>
    </template>
    
    <script>
        Polymer({
            is : 'cc-operorg-multi-combobox-field',
            properties : {
            	operUnitCd : {
            		type : Object,
            		value : function(){
            			return "";
            		}
            	},
            	disabled : {
            		type : Object,
            		value : function(){
            			return false;
            		}
            	},
            	selectedIndexes : {
            		type : Object,
            		value : function(){
            			return [];
            		}
            	},
            	selectedAll : {
            		type : Object,
            		value : function(){
            			return true;
            		}
            	},
            	enableCheckAll : {
            		type : Object,
            		value : function(){
            			return true;
            		}
            	},
                value : {
                    type : Object,
                    notify: true            //One-way binding (downward)
                /*<!-- changes to "{{value}}" propagate downward to "value" on child -->
                    <!-- changes to "value" are ignored by host due to square-bracket syntax -->*/
                },
                codes :  {
                    type : Object,
                    value : function(){
                        return {
                            operOrgCd : []
                        };
                    }
                }
            },
            
            //util.js에 promise를 처리를 위해 service구현
            service: function() {
                var me = this;
                UT.promise([me.$.codes]);
            },
            
            //최초 호출
            attached : function(){
                var me = this;
            }
        });
    </script>
</dom-module>
