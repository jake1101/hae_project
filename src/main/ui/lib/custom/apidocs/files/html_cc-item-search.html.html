/**
 * 품목 검색을 위한 input field 커스텀 태그
 *
 *     <cc-item-search value="{{value}}" multi-value="{{여러개받을value}}" class="w-200"></cc-item-search>
 *
 * @element cc-multi-text-field
 *
 *
 * @attribute value 해당 텍스트 필드 input value binding <br>
 * [string] [optional] 
 *
 *
 * @attribute placeholder placeholder <br>
 * [string] [optional] 
 *
 *  @attribute multi-value 멀티 value<br>
 * [string] [optional]
 *
  *  @attribute multi-check 멀티 체크 초기값 Y or N ,default N<br>
 * [string] [optional]
 *
 */
-->
<dom-module id="cc-item-search">
<!-- 
/**
 *
 *	@description : 품목코드 검색 custom tag
 *  <pre>
 * </pre>
 * @author : Yeon-u Kim
 * @FileName : 
 * @Since 2016. 5. 10.
 * @변경이력 : [2016. 5. 10.] Yeon-u Kim 최초작성
 */
-->
	<style>
	</style>
	
	<template>
       <div class="hbox">
	    	<sc-text-field 
	            class$="[[class]]"
	            value="{{value}}" placeholder$="[[placeholder]]" hidden="{{uiBind.multiable}}" maxLength="200">
	        </sc-text-field>
	        <sc-text-field 
	            class$="[[class]]" trigger-cls="multi-search"
	            value="{{displayValue}}" placeholder$="[[placeholder]]" readonly="true"
	            on-trigger-click="showDialog" hidden="{{!uiBind.multiable}}">
	        </sc-text-field>
	        <!-- <div class="trigger multi-search" on-trigger-click="showDialog" hidden="{{!uiBind.multiable}}"></div> -->
	        <sc-trigger-field trigger-cls="multi-search" on-trigger-click="onShowPrCateItemPopup"  hidden="{{!uiBind.multiable}}"></sc-trigger-field>
			<sc-checkbox-field 
	        	label="#{다중선택}" 
	        	on-click="onChangeMultiSelect" 
	        	input-value="{{multiCheck}}"
	        	checked-value="Y" un-checked-value="N"></sc-checkbox-field>
        </div>
	</template>
	
	<script>
	Polymer({
			is : 'cc-item-search',
			properties : {
				operOrgCd : {
					type : Object
				},
				value: {
                    type: Object,
                    value : function(){
                    	return {};
                    },
                    notify: true
                },
                multiValue: {
                	type: Array,
                    value: function() {
                        return [];
                    },
                    notify: true,
                    observer: "changedValue"
                },
                displayValue: {
                    type: String,
                    value: ""
                },
                multiCheck : {
					type : Object,
					value : function(){
						return "N";
					},
					notify:true
				},
                userEdit: { // 직접입력 여부
                    type: Boolean,
                    value: false
                },
                initValue: {
                    type: Array,
                    value: function() {
                        return [];
                    }
                },
                uiBind: {
                    type: Object,
                    value: function() {
                        return {
                            multiable : false
                        }
                    }
                }
			},
			//최초 호출
			attached : function(){
			},
			// 값 변경
            changedValue: function(value) {
                var me = this;
                if (UT.isArray(value)) {
                    var displayValue = "";
                    if (value[0]) {
                        displayValue = value[0];
                        if (value.length > 1) {
                            displayValue = this.translate("{0}외 {1}건",null,value[0], value.length - 1);
                        }
                    }
                    me.set("displayValue", displayValue);
                }
            },
          //다중선택 체크 변경
			onChangeMultiSelect : function(e){
				var me = this;
				me.set("uiBind.multiable",me.multiCheck === "Y");
				me.set("value","");
				me.set("displayValue", "");
                me.set("multiValue", []);
			},
			onShowPrCateItemPopup : function() {
				var me = this;
				if (!me.operOrgCd) {
					UT.alert(this.translate(MSG.PR1001)); //'운영단위를 선택하세요.'
					return false;
				}
				//popup에서 필요한 param 셋팅
				var param = {
					oper_org_cd : me.operOrgCd,
					purc_grp_typ : me.purcGrpTyp
				};
				//팝업 호출
				UT.popup("ep-item-search-list", me, 1000, 600,{
					'selected-items' : function(popup, e) {
						//선택  처리
						var item_cds= [];
						for(var i = 0;i<e.detail.length;i++){
							item_cds.push(e.detail[i].item_cd);
						}
						me.set("multiValue",item_cds);
						popup.close();
					},
					'load' : function(popup, e){
						//팝업에 load 함수를 호출한다.
						popup.getContent().load(param);
					}
				},{maximizable : true,title : me.translate("품목 검색")});
			}
		});
	</script>
</dom-module>
