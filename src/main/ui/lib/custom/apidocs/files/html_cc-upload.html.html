<dom-module id="cc-upload">

    <style>
    </style>
    
    <template>
        <sc-container id="uploadPanel" class$="[[class]]"></sc-container>
    </template>
    
    <script>
        Polymer({
            is: "cc-upload",
            properties: {
                uploadId: {
                    type: String,
                    value: ""
                },
                value: {
                    type: String,
                    value: "",
                    observer: "changedValue"
                },
                editable: {
                    type: Boolean,
                    value: false,
                    observer: "changedEditable"
                },
                views: {
                    type: String,
                    value: "list"
                },
                maxTotalFileCount: {
                    type: Number,
                    value: 0,
                    observer: "changedMaxTotalFileCount"
                },
                maxTotalFileSize: {
                    type: String,
                    value: "0",
                    observer: "changedMaxTotalFileSize"
                },
                allowExtension: {
                    type: String,
                    value: "",
                    observer: "changedAllowExtension"
                },
                width: {
                    type: String,
                    value: "100%"
                },
                height: {
                    type: String,
                    value: "200px"
                },
                showMoveButtons: {
                    type: Boolean,
                    value: false
                },
                imageOnly: {
                    type: Boolean,
                    value: false,
                    observer: "changedImageOnly"
                }
            },
            
            // 이미지 확장자 정의
            imageExtension: "jpg,jpeg,gif,png",
            
            // 화면 생성 완료
            attached: function() {
                var me = this;
            },

            // 업로더 service
            service: function() {
                var me = this;
                var promise = [
                    me.initUpload()
                ];
                Promise.all(promise).then(function(res) {
                    me.set("uploadId", res[0].uploadId);
                    me.fire("result", res[0].resolveName);
                });
            },
            
            // 업로더 생성
            initUpload: function() {
                var me = this;
                var options = {
                    editable: me.get("editable"),
                    showMoveButtons: me.get("showMoveButtons"),
                    views: me.get("views"),
                    maxTotalFileCount: me.get("maxTotalFileCount"),
                    maxTotalFileSize: me.get("maxTotalFileSize"),
                    width: me.get("width"),
                    height: me.get("height")
                };
                if (me.get("imageOnly")) {
                    options.extensionAllowOrLimit = "1";
                    options.extension = me.imageExtension;
                }
                if (typeof me.get("allowExtension") === "string" && me.get("allowExtension") !== "") {
                    options.extensionAllowOrLimit = "1";
                    options.extension = me.get("allowExtension");
                }
                return UPLOAD.createUpload(me.$.uploadPanel, options);
            },
            
            // 편집모드 설정
            changedEditable: function(editable) {
                var me = this;
                if (me.uploadId && typeof editable === "boolean") {
                    UPLOAD.setEditable(me.uploadId, editable);
                }
            },
            
            // 총 파일 개수 설정
            changedMaxTotalFileCount: function(count) {
                var me = this;
                if (me.uploadId && typeof count === "number") {
                    UPLOAD.setOptions(me.uploadId, {maxTotalFileCount: count});
                }
            },
            
            // 총 파일 용량 설정 - 단위 - B(byte), KB(kilobyte), MB(megabyte), GB(gigabyte), 0: 제한없음
            changedMaxTotalFileSize: function(size) {
                var me = this;
                if (me.uploadId && typeof size === "string") {
                    UPLOAD.setOptions(me.uploadId, {maxTotalFileSize: size});
                }
            },
            
            // 이미지만 업로드 가능 설정
            changedImageOnly: function(imageOnly) {
                var me = this;
                if (me.uploadId && typeof imageOnly === "boolean") {
                    UPLOAD.setOptions(me.uploadId, {extensionAllowOrLimit: "1", extension: imageOnly ? me.imageExtension : ""});
                }
            },
            
            // 허용할 확장자 설정
            changedAllowExtension: function(extension) {
                var me = this;
                if (me.uploadId && typeof extension === "string" && extension !== "") {
                    UPLOAD.setOptions(me.uploadId, {extensionAllowOrLimit: "1", extension: extension});
                }
            },
            
            // value 값
            changedValue: function(value) {
                var me = this;
                if (me.uploadId) {
                    if (UT.isEmpty(value)) {
                        UPLOAD.clearFiles(me.uploadId);
                    } else {
                        UPLOAD.onLoadFiles(me.uploadId, value);
                    }
                }
            },
            
            // 업로드 id 
            getId: function() {
                var me = this;
                return me.uploadId;
            }
            
        });
    </script>

</dom-module>
