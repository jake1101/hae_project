/**
* 다국어 변환
*
* @class I18N
*/
var I18N = {
    /**
     * 다국어 처리 instance
     */
    instance: null,

    /**
     * 다국어 변환
     *
     * @private
     * @method translate
     * @param  {string}
     * @return {string}
     */
    translate: function() {
        var me = this;
        if (me.instance === null) {
            me.instance = new SCI18n();
        }
        return me.instance.translate.apply(this, arguments);
    }
};
